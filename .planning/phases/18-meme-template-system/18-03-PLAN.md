```typescript
/**
 * Generate all death ending images - NO OVERLAY NEEDED
 * Templates like "I Declare Bankruptcy", "Straight to Jail" already have text
 */
async function generateAllDeathImages(): Promise<{ success: number; failed: number }> {
  const deathTypes: DeathType[] = ["BANKRUPT", "REPLACED_BY_SCRIPT", "CONGRESS", "FLED_COUNTRY", "PRISON", "AUDIT_FAILURE", "KIRK"];
  let success = 0;
  let failed = 0;
  
  for (const deathType of deathTypes) {
    const outputPath = path.join(
      process.cwd(),
      "public/images/deaths",
      `death-${deathType.toLowerCase()}-meme.webp`
    );
    
    const result = await generateDeathImage(deathType, outputPath);
    
    if (result.success) {
      success++;
      console.log(`✓ Death ${deathType}: ${result.template?.name}`);
    } else {
      failed++;
      console.warn(`✗ Death ${deathType}: ${result.error}`);
    }
  }
  
  return { success, failed };
}
```
