```typescript
// Get current personality from game state (import from where it's stored)
export async function playOnboarding(personality: Personality): Promise<void> {
  const path = getVoicePath(personality, 'onboarding');
  if (path) await playVoice(path);
}

export async function playVictory(personality: Personality): Promise<void> {
  const path = getVoicePath(personality, 'victory');
  if (path) await playVoice(path);
}

export async function playFailure(personality: Personality): Promise<void> {
  const path = getVoicePath(personality, 'failure');
  if (path) await playVoice(path);
}

export async function playFeedback(personality: Personality, feedbackType: 'paste' | 'debug' | 'install' | 'ignore'): Promise<void> {
  // Only Roaster has feedback voices
  if (personality !== 'roaster') return;
  
  const trigger = `feedback_${feedbackType}` as VoiceTrigger;
  const path = getVoicePath(personality, trigger);
  if (path) await playVoice(path);
}
```
