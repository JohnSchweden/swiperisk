1. **Create a helper function** to extract outcomes for a specific incident slug from HOS cards:

```typescript
/**
 * Extract outcomes for a specific incident from HOS cards using actual labels
 * Returns array of { label, labelSlug, lesson } for each choice
 */
function extractOutcomesForIncident(incidentSlug: string): Array<{ label: string; labelSlug: string; lesson: string }> {
  const outcomes: Array<{ label: string; labelSlug: string; lesson: string }> = [];
  
  for (const card of HEAD_OF_SOMETHING_CARDS) {
    if (!card.realWorldReference?.incident) continue;
    
    const cardIncidentSlug = slugifyIncident(card.realWorldReference.incident);
    if (cardIncidentSlug !== incidentSlug) continue;
    
    // onLeft choice
    outcomes.push({
      label: card.onLeft.label,
      labelSlug: slugifyLabel(card.onLeft.label),
      lesson: card.onLeft.lesson,
    });
    
    // onRight choice
    outcomes.push({
      label: card.onRight.label,
      labelSlug: slugifyLabel(card.onRight.label),
      lesson: card.onRight.lesson,
    });
  }
  
  return outcomes; // Should have 2 entries: left and right choices with their actual labels
}
```
