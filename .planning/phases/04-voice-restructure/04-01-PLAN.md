Replace with:
```typescript
const basePath = "/audio/voices";
const personalityDir = `${basePath}/${personality.toLowerCase().replace(/_/g, "")}`;

// Determine subfolder based on trigger prefix
function getSubfolder(trigger: string): string {
  if (trigger.startsWith("archetype_")) return "archetype";
  if (trigger.startsWith("death_")) return "death";
  if (trigger.startsWith("feedback_")) return "feedback";
  // Core triggers: onboarding, victory, failure
  if (["onboarding", "victory", "failure"].includes(trigger)) return "core";
  return ""; // Fallback to root (for backwards compatibility)
}

const subfolder = getSubfolder(trigger);
const filename = `${trigger}.wav`;
const filePath = subfolder 
  ? `${personalityDir}/${subfolder}/${filename}`
  : `${personalityDir}/${filename}`;
```

This maintains backwards compatibility if any files remain in root, but prefers subfolders.
