1. Add img.decode() usage to prevent jank:
   ```tsx
   const imgRef = useRef<HTMLImageElement>(null);

   const handleLoad = async (e: React.SyntheticEvent<HTMLImageElement>) => {
     const img = e.currentTarget;

     // Use decode() API to prevent jank before showing image
     if ('decode' in img) {
       try {
         await img.decode();
       } catch {
         // Decode failed but image may still be renderable
       }
     }

     setIsLoaded(true);
   };
   ```

2. Update the img element:
   - Add ref={imgRef}
   - Change onLoad to use handleLoad
