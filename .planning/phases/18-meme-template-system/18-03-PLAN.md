/**
 * Classify incident into pattern for template matching
 */
function classifyIncidentPattern(incident: string, outcome: string): string {
  const text = `${incident} ${outcome}`.toLowerCase();
  
  // Warning patterns
  if (text.includes("warn") || text.includes("red flag") || text.includes("ignore")) {
    return "ignored-warning";
  }
  // Escalating disaster
  if (text.includes("drift") || text.includes("cascad") || text.includes("chain")) {
    return "escalating-disaster";
  }
  // Bad decision
  if (text.includes("choice") || text.includes("decision") || text.includes("pick")) {
    return "bad-decision";
  }
  // AI overconfidence
  if (text.includes("ai") || text.includes("model") || text.includes("confidence")) {
    return "ai-overconfidence";
  }
  // Team failure
  if (text.includes("team") || text.includes("group") || text.includes("board")) {
    return "team-failure";
  }
  // Obvious outcome (default)
  return "obvious-outcome";
}
```

5. Add meme generation function:

/**
 * Generate a meme using template overlay
 */
async function generateMeme(
  incident: string,
  outcome: string,
  pattern: string
): Promise<{ success: boolean; message: string }> {
  const templates = getTemplatesByPattern(pattern);
  
  if (templates.length === 0) {
    return { success: false, message: `No templates for pattern: ${pattern}` };
  }
  
  // Pick random template from pattern
  const template = templates[Math.floor(Math.random() * templates.length)];
  
  // Generate punchline from outcome (truncate to ~10 words)
  const words = outcome.split(" ").slice(0, 10).join(" ");
  const punchline = words + (outcome.split(" ").length > 10 ? "..." : "");
  
  const buffer = await addTextToGif({
    templateId: template.id,
    text: punchline,
    fontSize: "20px",
    position: "bottom",
  });
  
  if (!buffer) {
    return { success: false, message: `Failed to generate meme: ${template.id}` };
  }
  
  const outputPath = path.join(
    process.cwd(),
    CATEGORY_DIR.incident,
    `${slugifyIncident(incident)}-meme.webp`
  );
  
  fs.mkdirSync(path.dirname(outputPath), { recursive: true });
  fs.writeFileSync(outputPath, buffer);
  
  return { success: true, message: `Meme generated: ${path.basename(outputPath)}` };
}
```

    if (result.success) {
      success++;
      console.log(`✓ ${result.message}`);

    }
  }

  console.log(`Generated: ${success}`);
  console.log(`Failed: ${failed}`);
  return;
}
