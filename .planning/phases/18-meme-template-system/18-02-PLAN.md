export interface DeathTemplate {
  id: string;

  url: string;
  text_zones: number;
  no_overlay?: boolean;
  primary?: boolean;
  animated?: boolean;

export type DeathType = "BANKRUPT" | "REPLACED_BY_SCRIPT" | "CONGRESS" | "FLED_COUNTRY" | "PRISON" | "AUDIT_FAILURE" | "KIRK";
export type Archetype = "PRAGMATIST" | "SHADOW_ARCHITECT" | "DISRUPTOR" | "CONSERVATIVE" | "BALANCED" | "CHAOS_AGENT" | "KIRK";

/**
 * Find a template by ID across all categories
 */
export function getTemplateById(id: string): TemplateInfo | undefined {
  for (const category of Object.values(templates.templates as Record<string, TemplateInfo[]>)) {
    const found = category.find((t) => t.id === id);
    if (found) {
      return {
        id: found.id,
        name: found.name,
        textZones: found.text_zones,
        bestFor: found.best_for,
        url: found.url || "",
      };
    }
  }
  return undefined;
}

/**
 * Get template for a death ending - NO OVERLAY NEEDED
 * Templates already convey the meaning (e.g., "Straight to Jail" for PRISON)
 */
export function getDeathTemplate(deathType: DeathType): DeathTemplate | undefined {
  const deathMapping = templates.death_mapping as Record<string, DeathTemplate[]>;
  const templatesForDeath = deathMapping[deathType];
  if (!templatesForDeath || templatesForDeath.length === 0) return undefined;
  
  // Return primary template (marked with primary: true) or first one
  const primary = templatesForDeath.find(t => t.primary);
  return primary || templatesForDeath[0];
}

/**
 * Get all templates for a death ending
 */
export function getDeathTemplates(deathType: DeathType): DeathTemplate[] {
  const deathMapping = templates.death_mapping as Record<string, DeathTemplate[]>;
  return deathMapping[deathType] || [];
}

/**
 * Get random death template
 */
export function getRandomDeathTemplate(deathType: DeathType): DeathTemplate | undefined {
  const templates = getDeathTemplates(deathType);
  if (templates.length === 0) return undefined;
  return templates[Math.floor(Math.random() * templates.length)];
}

/**
 * Search templates by keyword
 */
export function searchTemplates(query: string): TemplateInfo[] {
  const lowerQuery = query.toLowerCase();
  const results: TemplateInfo[] = [];
  
  for (const category of Object.values(templates.templates as Record<string, TemplateInfo[]>)) {
    for (const t of category) {
      if (t.name.toLowerCase().includes(lowerQuery) || 
          t.best_for.some((bf: string) => bf.toLowerCase().includes(lowerQuery))) {
        results.push({
          id: t.id,
          name: t.name,
          textZones: t.text_zones,
          bestFor: t.best_for,
          url: t.url || "",
        });
      }
    }
  }
  
  return results;
}

const TEMPLATE_CACHE_DIR = path.join(process.cwd(), "data/templates/cache");

/**
 * Fetch template image WITHOUT text overlay - for death endings and archetypes
 * Templates already have the text on them (e.g., "I Declare Bankruptcy")
 * Just fetch and return the image buffer
 */
export async function fetchTemplateNoOverlay(
  templateId: string,
  url: string
): Promise<Buffer | null> {
  try {
    const buffer = await fetchTemplateImage(templateId, url);
    return buffer;
  } catch (error) {
    console.error(`Error fetching template ${templateId}:`, error);
    return null;
  }
}

/**
 * Get template image as buffer — reads from local cache if available,
 * otherwise downloads from URL and caches to disk for future runs.
 */
async function fetchTemplateImage(templateId: string, url: string): Promise<Buffer> {
  const ext = url.split(".").pop()?.split("?")[0] || "jpg";
  const cachePath = path.join(TEMPLATE_CACHE_DIR, `${templateId}.${ext}`);

  if (fs.existsSync(cachePath)) {
    return fs.readFileSync(cachePath);
  }

  fs.mkdirSync(TEMPLATE_CACHE_DIR, { recursive: true });
  const buffer = await downloadImage(url);
  fs.writeFileSync(cachePath, buffer);
  return buffer;
}

/**
 * Download image from URL to buffer (raw, no cache)
 */
function downloadImage(url: string): Promise<Buffer> {
  return new Promise((resolve, reject) => {
    const protocol = url.startsWith("https") ? https : http;

    protocol.get(url, (res) => {
      if (res.statusCode === 301 || res.statusCode === 302) {
        const redirectUrl = res.headers.location;
        if (redirectUrl) {
          downloadImage(redirectUrl).then(resolve).catch(reject);
          return;
        }
        reject(new Error(`Redirect with no location: ${res.statusCode}`));
        return;
      }

      if (res.statusCode !== 200) {
        reject(new Error(`Failed to fetch: ${res.statusCode}`));
        return;
      }

      const chunks: Buffer[] = [];
      res.on("data", (chunk) => chunks.push(chunk));
      res.on("end", () => resolve(Buffer.concat(chunks)));
      res.on("error", reject);
    }).on("error", reject);
  });
}

/**
 * Add text to a meme template
 */
export async function addTextToMeme(
  options: OverlayOptions
): Promise<Buffer | null> {
  const {
    templateId,
    text,
    fontSize = "24px",
    fontColor = "white",
    strokeColor = "black",
    strokeWidth = 2,
    position = "bottom",
    getAsBuffer = true,
  } = options;

  // Get template URL from database
  const template = getTemplateById(templateId);
  if (!template) {
    console.error(`Template not found: ${templateId}`);
    return null;
  }

  const templateUrl = template.url;
  if (!templateUrl) {
    console.error(`No URL for template: ${templateId}`);
    return null;
  }

  try {
    // Fetch template — from local cache if available, otherwise download and cache
    const imageBuffer = await fetchTemplateImage(templateId, templateUrl);
    
    // Create TextOnGif instance with the image buffer
    const gif = new TextOnGif({
      file_path: imageBuffer,
      font_size: fontSize,
      font_color: fontColor,
      stroke_color: strokeColor,
      stroke_width: strokeWidth,
    });

    const result = await gif.textOnGif({
      text,
      alignment_x: "middle",
      alignment_y: position,
      get_as_buffer: getAsBuffer,
    });

    return result;
  } catch (error) {
    console.error(`Error adding text to meme ${templateId}:`, error);
    return null;
  }
}

/**
 * Add text to GIF (animated) - for GIF templates
 */
export async function addTextToGif(
  options: OverlayOptions
): Promise<Buffer | null> {
  // For animated GIFs, same implementation but handles multi-frame
  return addTextToMeme(options);
}

/**
 * Generate a death ending image - NO OVERLAY NEEDED
 * Templates like "Straight to Jail" already have the text
 */
export async function generateDeathImage(
  deathType: DeathType,
  outputPath?: string
): Promise<{ success: boolean; buffer?: Buffer; error?: string; template?: DeathTemplate }> {
  const template = getDeathTemplate(deathType);
  
  if (!template) {
    return { success: false, error: `No template found for death type: ${deathType}` };
  }
  
  // Fetch template image (no overlay)
  const buffer = await fetchTemplateNoOverlay(template.id, template.url);
  
  if (!buffer) {
    return { success: false, error: `Failed to fetch template: ${template.id}`, template };
  }
  
  // Save to file if path provided
  if (outputPath) {
    const dir = path.dirname(outputPath);
    fs.mkdirSync(dir, { recursive: true });
    fs.writeFileSync(outputPath, buffer);
  }
  
  return { success: true, buffer, template };
}

/**
 * Classify incident into pattern for template matching
 */
export function classifyIncidentPattern(incident: string, outcome: string): string {
  const text = `${incident} ${outcome}`.toLowerCase();
  
  // Warning patterns
  if (text.includes("warn") || text.includes("red flag") || text.includes("ignore") || text.includes("missed")) {
    return "ignored-warning";
  }
  // Escalating disaster
  if (text.includes("drift") || text.includes("cascad") || text.includes("chain") || text.includes("escalat")) {
    return "escalating-disaster";
  }
  // Bad decision
  if (text.includes("choice") || text.includes("decision") || text.includes("pick") || text.includes("select")) {
    return "bad-decision";
  }
  // AI overconfidence
  if (text.includes("ai") || text.includes("model") || text.includes("confidence") || text.includes("over")) {
    return "ai-overconfidence";
  }
  // Team failure
  if (text.includes("team") || text.includes("group") || text.includes("board") || text.includes("committee")) {
    return "team-failure";
  }
  // Obvious outcome (default)
  if (text.includes("obvious") || text.includes("predictable") || text.includes("expect")) {
    return "obvious-outcome";
  }
  // Overconfidence
  if (text.includes("自信") || text.includes("sure") || text.includes("certain")) {
    return "overconfidence";
  }
  // Data misuse
  if (text.includes("data") || text.includes("privacy") || text.includes("leak")) {
    return "data-misuse";
  }
  
  return "obvious-outcome";
}

/**
 * Generate a meme from incident info
 */
export async function generateMeme(
  options: MemeGenerateOptions
): Promise<{ success: boolean; buffer?: Buffer; error?: string }> {
  let template: TemplateInfo | undefined;
  
  // Determine template
  if (options.templateId) {
    template = getTemplateById(options.templateId);
  } else if (options.pattern) {
    template = getRandomTemplateByPattern(options.pattern);
  } else if (options.incident && options.outcome) {
    const pattern = classifyIncidentPattern(options.incident, options.outcome);
    template = getRandomTemplateByPattern(pattern);
  }
  
  if (!template) {
    return { success: false, error: "No template found" };
  }
  
  // Generate text
  let text = options.customText || "";
  if (!text && options.outcome) {
    // Truncate outcome to punchline
    const words = options.outcome.split(" ").slice(0, 8).join(" ");
    text = words + (options.outcome.split(" ").length > 8 ? "..." : "");
  }
  
  if (!text) {
    return { success: false, error: "No text provided" };
  }
  
  // Generate meme
  const buffer = await addTextToMeme({
    templateId: template.id,
    text,
    fontSize: options.fontSize || "20px",
  });
  
  if (!buffer) {
    return { success: false, error: "Failed to generate meme" };
  }
  
  // Write to file if path provided
  if (options.outputPath) {
    const dir = path.dirname(options.outputPath);
    fs.mkdirSync(dir, { recursive: true });
    fs.writeFileSync(options.outputPath, buffer);
  }
  
  return { success: true, buffer };
}

/**
 * Batch process multiple memes
 */
export async function batchGenerateMemes(
  tasks: MemeGenerateOptions[]
): Promise<Array<{ success: boolean; templateId?: string; error?: string }>> {
  const results = [];
  
  for (const task of tasks) {
    const result = await generateMeme(task);
    results.push({
      success: result.success,
      templateId: task.templateId,
      error: result.error,
    });
    
    // Rate limiting
    if (tasks.length > 1) {
      await new Promise((r) => setTimeout(r, 1000));
    }
  }
  
  return results;
}
