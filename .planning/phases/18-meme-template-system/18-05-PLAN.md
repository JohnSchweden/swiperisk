```typescript
import TextOnGif from "text-on-gif"
import * as fs from "node:fs"
import * as path from "node:path"
import { downloadMeme } from "./meme-downloader"

// Templates whose imagery already contains readable text or captions.
// Do NOT add overlay to these — it would obscure the existing content.
const TEMPLATES_WITH_EXISTING_TEXT = new Set([
  "waiting-skeleton",
  "zuckerberg-congress",
  "kirk-jailbreak-llm",
  "success-kid",
  "grus-plan",
  "disaster-girl",
  "change-my-mind",
  "theyre-same-picture",
  "this-is-fine",
  "never-think-out-of-box",
])

/** Returns true if the template already has pre-existing caption text. */
export function templateHasText(templateId: string): boolean {
  return TEMPLATES_WITH_EXISTING_TEXT.has(templateId)
}

/**
 * Add a punchline text overlay to a meme template.
 * ONLY call after confirming templateHasText() returns false.
 * Returns null if overlay fails; caller should fall back to as-is.
 */
export async function addPunchlineOverlay(
  templateId: string,
  templateUrl: string,
  punchline: string,
  outputPath?: string,
): Promise<Buffer | null> {
  try {
    const localPath = await downloadMeme(templateId, templateUrl)
    const gif = new TextOnGif({
      file_path: localPath,
      font_size: "24px",
      font_color: "white",
      stroke_color: "black",
      stroke_width: 2,
    })

    const result = await gif.textOnGif({
      text: punchline,
      alignment_x: "middle",
      alignment_y: "bottom",
      get_as_buffer: true,
    })

    if (outputPath && result) {
      fs.mkdirSync(path.dirname(outputPath), { recursive: true })
      fs.writeFileSync(outputPath, result as Buffer)
    }

    return (result as Buffer) ?? null
  } catch (error) {
    console.error(`meme-overlay: error adding overlay to ${templateId}:`, error)
    return null
  }
}

/**
 * Generate a meme with punchline text.
 *
 * Decision flow:
 *   1. If template has pre-existing text → copy as-is, no overlay
 *   2. Otherwise → add punchline overlay via text-on-gif
 *   3. On overlay failure → fall back to template as-is
 *
 * Returns { hasExistingText, buffer?, outputPath? }
 */
export async function generateMemeWithPunchline(
  templateId: string,
  templateUrl: string,
  punchline: string,
  outputPath?: string,
): Promise<{ hasExistingText: boolean; buffer?: Buffer; outputPath?: string }> {
  if (templateHasText(templateId)) {
    const localPath = await downloadMeme(templateId, templateUrl)
    if (outputPath) {
      fs.mkdirSync(path.dirname(outputPath), { recursive: true })
      fs.copyFileSync(localPath, outputPath)
      return { hasExistingText: true, outputPath }
    }
    const buffer = fs.readFileSync(localPath)
    return { hasExistingText: true, buffer }
  }

  const buffer = await addPunchlineOverlay(templateId, templateUrl, punchline, outputPath)

  if (!buffer) {
    // Fallback: use template without overlay
    const localPath = await downloadMeme(templateId, templateUrl)
    const fallback = fs.readFileSync(localPath)
    if (outputPath) {
      fs.mkdirSync(path.dirname(outputPath), { recursive: true })
      fs.writeFileSync(outputPath, fallback)
    }
    return { hasExistingText: false, buffer: fallback, outputPath }
  }

  return { hasExistingText: false, buffer, outputPath }
}
```

Then create scripts/meme-overlay.ts:

```typescript
#!/usr/bin/env bun
/**
 * Add punchline text to meme templates.
 * Checks for pre-existing text first — skips overlay if template is already captioned.
 *
 * Usage:
 *   bun run scripts/meme-overlay.ts --template allow-claude-use --punchline "You enabled Claude" --output out.jpg
 *   bun run scripts/meme-overlay.ts --incident "AI leaked data" --outcome "Regulators fined $40M" --output out.jpg
 */

import { generateMemeWithPunchline, templateHasText } from "../lib/meme-overlay"
import templates from "../data/templates/meme-templates.json"

const args = process.argv.slice(2)
const get = (flag: string) => {
  const i = args.indexOf(flag)
  return i !== -1 ? args[i + 1] : undefined
}

const templateId = get("--template") ?? ""
const punchline = get("--punchline") ?? ""
const incident = get("--incident") ?? ""
const outcome = get("--outcome") ?? ""
const outputPath = get("--output") ?? ""

function makePunchline(text: string): string {
  const words = text.split(" ")
  return words.length > 10 ? words.slice(0, 10).join(" ") + "..." : text
}

function findTemplateUrl(id: string): string | undefined {
  const allEntries = [
    ...Object.values(templates.deaths).flatMap((d) => [d.static, d.gif]),
    ...Object.values(templates.archetypes).flatMap((a) => [a.static, a.gif]),
    ...Object.values(templates.feedbackOutcomes),
  ]
  return (allEntries as Array<{ id: string; url: string }>).find((e) => e?.id === id)?.url
}

async function main() {
  if (!templateId && !incident) {
    console.log(`
Usage:
  bun scripts/meme-overlay.ts --template <id> --punchline "text" [--output path]
  bun scripts/meme-overlay.ts --incident "what happened" --outcome "result" [--output path]

Flags:
  --template   Template ID (e.g. i-too-like-to-live-dangerously)
  --punchline  Text to overlay
  --incident   Incident description (auto-generates punchline from --outcome)
  --outcome    Outcome description
  --output     Output file path (optional; prints to console if omitted)

Behaviour: if the template already has pre-existing text, it is used as-is with no overlay added.
`)
    return
  }

  const resolvedPunchline = punchline || makePunchline(outcome)
  const resolvedId = templateId

  const templateUrl = findTemplateUrl(resolvedId)
  if (!templateUrl) {
    console.error(`Template not found: ${resolvedId}`)
    process.exit(1)
  }

  console.log(`Template: ${resolvedId}`)
  console.log(`Has existing text: ${templateHasText(resolvedId)}`)
  console.log(`Punchline: "${resolvedPunchline}"`)

  const result = await generateMemeWithPunchline(resolvedId, templateUrl, resolvedPunchline, outputPath || undefined)

  if (result.hasExistingText) {
    console.log("Template has pre-existing text — used as-is, no overlay added.")
  } else {
    console.log("Overlay applied.")
  }

  if (result.outputPath) {
    console.log(`Saved to: ${result.outputPath}`)
  } else if (result.buffer) {
    console.log(`Buffer: ${result.buffer.length} bytes (no --output path specified)`)
  }
}

main().catch(console.error)
```
