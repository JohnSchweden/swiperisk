   ```tsx
   const imgRef = useRef<HTMLImageElement>(null);
   
   const handleLoad = async (e: React.SyntheticEvent<HTMLImageElement>) => {
     const img = e.currentTarget;
     
     // Use decode() API to prevent jank before showing image
     if ('decode' in img) {
       try {
         await img.decode();
       } catch {
         // Decode failed but image may still be renderable
       }
     }
     
     setIsLoaded(true);
   };
   ```

2. Update the img element:
   - Add ref={imgRef}
   - Change onLoad to use handleLoad

3. Optional: Add intersection observer for advanced lazy loading control:
   ```tsx
   const [isInViewport, setIsInViewport] = useState(false);
   const containerRef = useRef<HTMLDivElement>(null);
   
   useEffect(() => {
     if (!containerRef.current || !('IntersectionObserver' in window)) {
       setIsInViewport(true);
       return;
     }
     
     const observer = new IntersectionObserver(
       ([entry]) => {
         if (entry.isIntersecting) {
           setIsInViewport(true);
           observer.disconnect();
         }
       },
       { rootMargin: '50px' }
     );
     
     observer.observe(containerRef.current);
     return () => observer.disconnect();
   }, []);
   ```

4. Only render img when in viewport:
   - Conditionally render img only if isInViewport
   - This provides enhanced lazy loading beyond native loading="lazy"

5. Keep native loading="lazy" as fallback/primary mechanism.
